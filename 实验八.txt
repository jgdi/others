struct bittree *init(struct bittree *BT)
{
	char ch;
	scanf("%c",&ch);
	if(ch==' ')
	{
		BT=NULL;
	}
	else
	{
		BT=new bittree;
		BT->data=ch;
		BT->lchild=init(BT->lchild);
		BT->rchild=init(BT->rchild);
	}
	return BT;
}
void FirstShow(struct bittree *BT)
{
	if(BT!=NULL)

		printf("%c",BT->data);
		FirstShow(BT->lchild);
